{% endif %} Note: "block", "item", and "enchant" types are strings. Simply prepend "/json" to get the new path to request for more information.

/json/block/<block-name>

{ "name": string, "drops": {item, int?}[], // The int is the drop chance as a "1 in" value "dew_to": block?, "dewed_from": block[], "hardness": int?, "required_breaking_power": int?, "id": string, "layers": string[] }

/json/item/<item-name>

{ "name": string, "forge_result": item?, "forged_from": item?, "crafting_tier": string?, "crafting_recipes": item[][][], "special_obtain": string[], "special_uses": string[], "museum_position": int[]?, "sell_price": int?, "rarity": string?, "radiation": int?, "mining_speed": int?, "breaking_power": int?, "accessory_use": string?, "accessory_slots": int?, "module_effect": string?, "module_slots": int?, "modular_armor_slots": int?, "modular_armor_capacity": int?, "max_depth": int?, "oxygen_capacity": int?, "lore": string[], "item": { "id": string, "enchanted"?: boolean, // Only present if true "cmd": int, // The custom model data "color"?: string, // A hex color; only present when the item is colorable "value"?: string, // A head value string; only present when the item is a head "water"?: boolean, // If the potion is water; only present if true }, "dropped_from": {block, int?}, // The int is the drop chance as a "1 in" value "crafting_uses": item[], "enchantment_uses": {enchant, int}[] // The int is the level }

/json/enchant/<enchant-name>

{ "name": string, "effect": string, "location": string, "requirements": item[][] }

/json/artifact/<artifact-name>

{ "name": string, "number": int, "lore": string[], "color": string // A hex color for the name }

/json/items

This returns a list of the below structures.
{ "name": string, "id": string, "enchanted": boolean, "rarity": string, "path": string, "lore": string[] }
